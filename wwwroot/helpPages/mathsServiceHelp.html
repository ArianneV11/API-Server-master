Get : Maths endpoint
List of possible query strings:

? op = + & x = number & y = number
return {"op":"+","x":number,"y":number,"value": x + y}

? op = - & x = number & y = number
return {"op":"-","x":number,"y":number,"value": x - y}

? op = * & x = number & y = number
return {"op":"*","x":number,"y":number,"value": x * y}

? op = / & x = number & y = number
return {"op":"/","x":number,"y":number,"value": x / y}

? op = % & x = number & y = number
return {"op":"%","x":number,"y":number,"value": x % y}

? op = ! & n = integer
return {"op":"!","n":number,"value": n!}

? op = p & n = integer
return {"op":"p","n":number,"value": true if n is a prime number}

? op = np & n = integer
return {"op":"np","n":number,"value": nth prime number}
